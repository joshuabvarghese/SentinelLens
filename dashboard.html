async function loadData() { try { const response = await fetch('monitoring_results.json'); const data = await response.json(); // Update the "Last Updated" timestamp document.getElementById('timestamp').textContent = new Date(data.timestamp).toLocaleString(); // Render the cards for CPU, Memory, and Disk const container = document.getElementById('checks-container'); container.innerHTML = data.checks.map(check => `

${check.name.toUpperCase()}

${check.message}

${check.status}
`).join(''); } catch (error) { console.error('Error loading monitoring data:', error); } } // Auto-refresh logic setInterval(loadData, 30000); loadData();
